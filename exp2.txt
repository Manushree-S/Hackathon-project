# Create and enter a new directory
mkdir Exp2
cd Exp2

# Initialize a new Git repository
git init

# Create a text file with content
echo "Project File" > project.txt

# Verify file content and check repository status
cat project.txt
git status

# Stage the new file for commit
git add .

# Check status again to see the staged file
git status

# Commit the changes with a message
git commit -m "First Commit"

# Link the local repository to a remote GitHub repository
git remote add origin https://github.com/Manushree-S/Exp2.git

# Verify the remote connection
git remote -v

# Rename the current branch to 'main'
git branch -M main

# Push the 'main' branch to the 'origin' remote and set it as the default tracking branch
git push -u origin main
